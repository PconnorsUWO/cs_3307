213void Board::revealCell(int row, int col)
214{
215 if (row < 0 || row >= m_rows || col < 0 || col >= m_cols) {
216 return;
217 }
218
219 // Queue for BFS traversal.
220 std::queue<std::pair<int, int>> toProcess;
221 toProcess.push({row, col});
222
223 while (!toProcess.empty()) {
224 auto [currentRow, currentCol] = toProcess.front();
225 toProcess.pop();
226
227 Cell &currentCell = m_cells[currentRow][currentCol];
228
229 // Skip if the cell is already revealed or is a mine.
230 if (currentCell.isRevealed() || currentCell.isMine()) {
231 continue;
232 }
233
234 // Reveal the cell.
235 currentCell.setIsRevealed(true);
236
237 // If no adjacent mines, add neighbors to the queue.
238 if (currentCell.adjacentMines() == 0) {

240 for (Cell *neighbor : adj) {

242 int neighborRow = neighbor->row();
243 int neighborCol = neighbor->col();

245 }
246 }
247 }
248 }
